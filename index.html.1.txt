        let currentStep = 1;
        const totalSteps = 5;

        // SYSTEM A: STEP TRANSITION TIMELINES WITH GSAP
        function navigateToStep(nextStep) {
            const currentEl = document.getElementById(`step-${currentStep}`);
            const nextEl = document.getElementById(`step-${nextStep}`);
            
            // Progress Bar Update Calculation
            const progressPercent = (nextStep / totalSteps) * 100;
            document.getElementById('progress-bar-fill').style.width = `${progressPercent}%`;

            gsap.to(currentEl, {
                opacity: 0,
                y: -30,
                scale: 0.95,
                duration: 0.5,
                ease: "power2.inOut",
                onComplete: () => {
                    currentEl.classList.remove('active');
                    nextEl.classList.add('active');
                    
                    // Stagger content components inward on next step activation
                    gsap.fromTo(nextEl.children, 
                        { opacity: 0, y: 30 },
                        { opacity: 1, y: 0, duration: 0.6, stagger: 0.15, ease: "power3.out" }
                    );
                }
            });

            currentStep = nextStep;
        }

        // SYSTEM B: POLAROID 3D MOUSE PARALLAX CONTROLLER
        const polaroid = document.getElementById('polaroid');
        if(polaroid) {
            document.addEventListener('mousemove', (e) => {
                if (!document.getElementById('step-4').classList.contains('active')) return;
                
                const rect = polaroid.getBoundingClientRect();
                const x = e.clientX - (rect.left + rect.width / 2);
                const y = e.clientY - (rect.top + rect.height / 2);
                
                // Scale structural rotations to comfort levels
                const rotateX = -y / 15;
                const rotateY = x / 15;
                
                gsap.to(polaroid, {
                    rotationX: rotateX,
                    rotationY: rotateY,
                    transformPerspective: 1000,
                    duration: 0.3,
                    ease: "power1.out"
                });
            });

            // Reset matrix coordinates on mouse leave
            polaroid.addEventListener('mouseleave', () => {
                gsap.to(polaroid, { rotationX: 0, rotationY: 0, duration: 0.5, ease: "power2.out" });
            });
        }
